## Install

```bash
npm install --save react-dropdown
```

## Usage

```jsx
import React from 'react';
import Dropdown from "react-dropdown";
import 'react-dropdown/dist/index.scss';

const App = () => {

  return <Dropdown 
            setValue={func}
            list={[datas]}
            resetDrop={bool}
        />
}

export default App
```
